alter table nfce add hora varchar(8);
alter table nfce add xml_cancelamento varchar(255);



create or alter procedure NFCE_INSERT (
    PNUMERO integer,
    PDATA date,
    PHORA varchar(8),
    PTOTAL numeric(15,2),
    PCLIENTE varchar(255),
    PCHAVE varchar(255),
    PXML varchar(255),
    PSITUACAO integer,
    PTROCO numeric(15,2))
as
BEGIN

INSERT INTO nfce  (numero, data, hora, total, cliente,chave, xml, situacao, troco)
values  (:pnumero, :pdata, :phora, :ptotal, :pcliente, :pchave, :pxml, :psituacao, :ptroco);

END;

GRANT INSERT ON NFCE TO PROCEDURE NFCE_INSERT;

GRANT EXECUTE ON PROCEDURE NFCE_INSERT TO SYSDBA;
